\unit{ $ Statistic models $ }
{
	\introduction
	{ 
		$introduction\\$ 
	}

	\definition{ $ Statistical model $ }
	{
		\letbe
		{
			\Omega $ set $.
			\Ac $ sigma-algebra over $ \Omega.
			\Pc \subset \functionSpace{ \Omega }{ \R }
		}
		\then{ ( \Omega, \Ac, \Pc ) }{ $ a statistical model $ }
		{
			\all{ f \in \Pc }
			{
				f $ probability distribution $
			}
		}
	}
	


	\definition{ $ Parametrized $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ statistical model $.
			\Theta \subset \R^d
		}
		\then{ ( \Omega, \Ac, \Pc ) }{ $ parametrized by $ \Theta }
		{
			\Pc = \family*{ P_\theta }{ \theta \in \Theta }
		}
	}
	
	
	\definition{ $ Likelihood $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ statistical model parametrized by $ \Theta
		}
		\name{ $ likelihood function of $ ( \Omega, \Ac, \Pc ) }
		{
			\definedFunction{ L }{ \Omega \times \Theta }{ \R^+ }{ (x,\theta) }{ P_\theta(x) }
		}
	}
	


	\definition{ $ Exponential model $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ statistical model parametrized by $ \Theta.
			L $ likelihood function of $ ( \Omega, \Ac, \Pc )
		}
		\then{ ( \Omega, \Ac, \Pc ) }{ $ exponential $ }
		{
			\ex{ \family{ f_i }{ i }[ 1 ][ r ], f \in \functionSpace{ \Omega }{ \R } }{ \all{ i \in \indexes{ 1 }{ r } }
			{
				f_i,f $ measurable $
			} }.
			\ex{ \family{ \phi_i }{ i }[ 1 ][ r ], \phi \in \functionSpace{ \theta }{ \R } }{ \all{ i \in \indexes{ 1 }{ r } }
			{
				\phi_i, \phi $ measurable $
			} }.
			\all{ x \in \Omega }
			{
				\all{ \theta \in \Theta }
				{
					L( x, \theta ) = \exp( \summation{ f_i(x)\phi_i(\theta) }{ i }[ 1 ][ r ] + f(x) + \phi(\theta) )
				}
			}
		}

		
		
		
		
	}
