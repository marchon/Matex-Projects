\unit{ $ Estimation $ }
{

	\definition{ $ Statistic $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ m-D statistical model parametrized by $ \Theta.
			( M, \Sigma ) $ measurable space $.

			\function{ X }{ \Omega }{ M } $ random variable $.
			\function{ f }{ M }{ \R^m }
		}
		\then{ f }{ $ a statistic $ }
		{
			f $ measurable $
		}
		\denote
		{
			f \as T
		}
	}


	\definition{ $ Sufficiency $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ m-D statistical model parametrized by $ \Theta.
			( M, \Sigma ) $ measurable space $.
			\function{ X }{ \Omega }{ M } $ random variable $.
			\function{ T }{ M }{ \R^m } $ statistic $
		}
		\then{ T }{ $ sufficient $ }
		{
			\all{ \theta_1,\theta_2 \in \Theta }
			{
				\all{ x \in M }
				{
				 	\all{ t \in \R^m }
				 	{
				 		P_{\theta_1}(X = x \with T \circ X = t ) = P_{\theta_2}( X = x \with T \circ X = t )
				 	}
				}
			}
		}
	}


	\definition{ $ Estimator $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ statistical model parametrized by $ \Theta.
			( M, \Sigma ) $ measurable space $.
			\function{ X }{ \Omega }{ M } $ random variable $.
			\function{ T }{ M }{ \R^k } $ statistic $.
			\theta \in \Theta
		}
		\then{ T }{ $ an estimator of $ \theta }
		{
			T $ approaches $ \theta
		}
	}
	
	
	\definition{ $ Loss function $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ statistical model parametrized by $ \Theta.
			( M, \Sigma ) $ measurable space $.
			\function{ X }{ \Omega }{ M } $ random variable $.
			\function{ T }{ M }{ \R^k } $ estimator $.
			\function{ W }{ \R^k \times \Theta }{ \R^+ }
		}
		\then{ W }{ $ a loss function $ }
		{
			W(\theta,\theta) = 0
		}
	}
	
	
	\definition{ $ Risk function $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ statistical model parametrized by $ \Theta.
			( M, \Sigma ) $ measurable space $.
			\function{ X }{ \Omega }{ M } $ random variable $.
			\function{ T }{ M }{ \R^k } $ estimator $.
			\function{ W }{ \R^k \times \Theta }{ \R^+ } $ loss function $
		}
		\name{ $ risk function $ }
		{
			\definedFunction{ R_T }{ \Theta }{ \R^+ }{ \theta }{ E_\theta(W(T,\theta)) }
		}
	}

	\definition{ $ UMV $ }
	{
		\letbe
		{
			statements.
		}
		\then{ item }{ $ a/an entity $ }
		{
			conditions.
		}
		\denote
		{
			property \as notation.
		}
	}
	
	
	\definition{ $ Same conditions $ }
	{
		\letbe
		{
			( \Omega, \Ac, \Pc ) $ parametric statistical model $.
			\function{ X }{ \Omega }{ \R } $ random variable $.
			\Theta \subset \R $ interval $.
			\chi_F $ real estimator with integrable quadratic $.
		}
		\then{ item }{ $ a/an entity $ }
		{
			\all{ \theta \in \Theta }
			{
				\ex{ \function{ h }{ \R }{ \R } }
				{
					h \geq 0.
					h $ integrable $.
					\ex{ \Uc \subset \R }
					{
						\theta \in \Uc.
						|T(x) \partialderivative{ L(x,\theta) }{ \theta }| \leq h
					}
				}
			}
		}
	}


	\definition{ $ Efficient $ }
	{
		\letbe
		{
			$mismas condiciones$
		}
		\then{ T }{ $ efficient $ }
		{
			Var_\theta T = \frac{ g'(\theta)^2 }{ I(\theta) }
		}
	}
}
