\definition{ $ Incremental quotient $ }
{
	\letbe
	{
		\Uc \subset \C $ open $.
		\function{ f }{ \Uc }{ \C }.
		p \in \Uc
	}
	\name{ $ incremental quotient of $ f $ in $ p }
	{
		\limit{ \frac{ f(z) - f(p) }{ z-p } }{ z }[ p ]
	}
	\denote
	{
		f'(p)
	}
}





\definition{ $ Holomorphic function $ }
{
	\letbe
	{
		\Uc \subset \C $ open $.
		\function{ f }{ \Uc }{ \C }.
		p \in \Uc
	}
	\then{ f }{ $ holomorphic over $ p $ $ }
	{
		\exists f'(p)
	}
	\then{ f }{ $ holomorphic over $ U $ $ }
	{
		\all{ p \in \Uc }
		{
			\exists f'(p)
		}
	}
	\denote
	{
		\set{ \function{ f }{ \Uc }{ \C } \with f $ holomorphic over $ \Uc } \as \Hc(\Uc).
		f \in \Hc(\C) \as f $ entire $
	}
}


\definition{ $ Real components $ }
{
	\letbe
	{
		f \in \Hc(\Uc)
	}
	\name{ $ first real component of $ f }
	{
		\definedFunction{ f_{Re} }{ \Uc }{ \R }{ z }{ Re( f(z) ) }
	}
	\name{ $ second real component of $ f }
	{
		\definedFunction{ f_{Im} }{ \Uc }{ \R }{ z }{ Im( f(z) ) }
	}
}

\definition{ $ Real dual $ }
{
	\letbe
	{
		f \in \Hc( \Uc ).
		f_{Re}, f_{Im} $ real components of $ f
	}
	\name{ $ real dual of $ f }
	{
		\definedFunction{ f_{\R^2} }{ \R^2 }{ \R^2 }{ (x,y) }{ ( f_{Re}( x + yi ), f_{Im}( x + yi ) ) }
	}
	\denote
	{
		property \as notation.
	}
}
