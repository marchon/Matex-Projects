	\definition{ $ Cauchy-Riemman equations $ }
	{
		\letbe
		{
			\definedFunction{ f }{ \R^2 }{ \R^2 }{ x }{ (u(x),v(x)) }
		}
		\then{ f }{ $ satisfies the Cauchy-Riemman equations $ }
		{
			u_x = v_y.
			u_y = - v_x
		}
		\denote
		{
			u_x + iv_x \as f_x.
			u_y + iv_y \as f_y
		}
	}
	
	
	\proposition{ $ Cauchy-Riemman $ }
	{
		\letbe
		{
			\function{ f }{ \R^2 }{ \R^2 } $ satisfies CR $
		}
		\holds
		{
			f_x = f'(z).
			f_y = -if_y.
			\partialderivative{ z }{ f(z) } = 0
		}
		\demonstration
		{
			f_x = u_x + iv_x = a + bi = f'(z).
			f_y = i(a+ib) = if'(z).
			f'(z) = -if_y.
			$notation$
		}
	}

	\definition{ $ Plane arc $ }

		}
	}

	\proposition{ $ arc images $ }
	{
		\letbe
		{
			f \in \Hc(\Uc).
			\gamma $ differentiable plane arc $ \suchthat \gamma \subset \Uc.
			\sigma = f(\gamma).
			z_0 = \gamma(0)
		}
		\holds
		{
			\sigma' = f'(\gamma)\gamma'.
			\gamma'(0) \neq 0 \imp f'(z_0) \neq 0.
			\sigma'(0) = f'(z_0)\gamma'(0).
			|\sigma'(0)| = |f'(z_0)||\gamma'(0)|.
			arg \sigma'(0) = arg \gamma'(0) + arg f(z_0).
			$ f aplica una homotecia mas una rotacion constante a todos los \\ vectores tangentes que salen de z0 $
		}
		\demonstration
		{
			$obvio$
		}
	}


	\definition{ $ Conformalism $ }
	{
		\letbe
		{
			f $ function $
			z_0 \in \Uc
		}
		\then{ f }{ $ conform in $ z_0 }
		{
			\all{ \function{ \gamma }{ I }{ \R^2 }}[ \gamma $ differentiable $ ]
			{
				\gamma(0) = z_0 \gamma'(0) \neq 0.
				\sigma := f(\gamma).
				arg \sigma'(0) - arg \gamma'(0) $ constant $.
				|\frac{ \sigma'(0) }{ \gamma'(0) }| $ constant $
			}
		}
	}
