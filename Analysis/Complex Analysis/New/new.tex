	\definition{ $ Absolutely convergent $ }
	{
		\letbe
		{
			\sum{ c_n }{ n }[ 0 ] $ series $
		}

	\proposition{ $ Absolutely convergent are convergent $ }
	{
		\letbe
		{
			\sum{ c_n }{ n }[ 0 ] $ absolutely convergent $
		}
		\holds
		{
			\sum{ c_n }{ n }[ 0 ] $ convergent $
		}
		\demonstration
		{
			\be{ S_k }{ \sum{ c_n }{ n }[ 0 ][k] }.

			\all{ m \in \N }[ m < k ]
			{
				|S_k - S_m | = | \sum{ c_n }{ n }[ m+1 ][k] | \leq \sum{ |c_n| }{ n }[ m+1 ][k].

				\leq \sum{ |c_n| }{ n }[ m+1 ] \convergesto 0.

				|S_k - S_m | \convergesto 0 \imp (S_k)_k $ convergent $ \imp \sum{ c_n }{ n }[ 0 ] $ convergent $
			}
		}
	}

	\proposition{ $ Power series theorem $ }
	{
		\letbe
		{
			\summation{ a_nc^n }{ n }[ 0 ] $ power series $
		}
		\holds
		{
			|z-a| < R \imp $ absolutely convergent $.
			|z-a| > R \imp $ divergent $.

			$ convergent in $ D(a,R).

			\definedFunction{ f }{ \C }{ \C }{ z }{ \summation{ c_n(z-a)^n }{ n }[ 0 ] } \in \Hc(D(a,R)).

			\all{ z \in \C }
			{
				f'(z) = \summation{ nc_n(z-a)^{n-1} }{ n }[ 0 ] $ convergent $ 
			}.

			$ convergence radius of $ f' = $ convergence radius of $ f
		}
		\demonstration
		{
			 \all{ z \in \C }[ |z-a| < R ]
			 {
			 	$ Root test over $ \summation{ |c_n||z-a|^n }{ n }[ 0 ]

			 	\limit{ (|c_n||z-a|^n)^{\frac{ 1 }{ n }) }{ n } =
			 	|z-a| \limit{ |c_n|^{\frac{ 1 }{ n }} }{ n } = \frac{ |z-a| }{ R } < 1

			 	$ Root test $ \imp $ absolutely convergent $.

			 }

			 \all{ z \in \C }[ |z-a| < R ]
			 {
			 	\all{ \rho \in \R }[ |z-a| < \rho < R ]
			 	{
			 		\frac{ 1 }{ \rho } < \frac{ 1 }{ R }.

			 		\limit{ |c_n|^{1/n} }{ n } = \frac{ 1 }{ R } \imp

			 		$ exists partial of $ |c_n|^{1/n} $:..$.

			 		$ General term test $ \imp $ divergent $
			 	}
			}
		}
		}
