	\proposition{ $ Characterization of sella-node bifurcation points $ }
	{
		\letbe
		{
			I \subset \R
			\family*{ \function{ f_\lambda }{ I }{ I } }{ \lambda \in \Lambda }.
			x \in I
		}
		\holds
		{
			x $ SN bifurcation point $ \ifandonlyif f(x) = x, f'(x) = 1.

			\partialderivative{ f }{ \lambda } \neq 0.

			f''(x) \neq 0
		}
		\demonstration
		{
			demonstration.
		}
	}
	
	
	
	\example{ $ Quadratic function bifurcations $ }
	{
		\letbe
		{
			\definedFunction{ f }{ \R }{ \R }{ x }{ a - x^2 }.
			( M, T, f_c ) $ dynamical system family $
		}
		\is{ f }{ $ bifurcates in -1/4 $ }
		{
			f_{-\frac{ 1 }{ 4 }}(x) = x \ifandonlyif x = -\frac{ 1 }{ 2 }.

			f'_{-\frac{ 1 }{ 4 }}(x) = -2x.

			f'_{-\frac{ 1 }{ 4 }}(-\frac{ 1 }{ 2 }) = 1.

			\partialderivative{ f }{ a } = 1 \neq 0.

			\partialderivative{ f }{ x^2 } = -2 \neq 0.

			sgn( 1 * -2 ) = - \imp -\frac{ 1 }{ 2 } $ SN $
		}
	}
	
	
	\definition{ $ Pitchfork bifurcation $ }
	{
		\letbe
		{
			( M, T, f_\lambda ) $ dynamical system family $.
			x \in M
		}
		\then{ x }{ $ pitchfork bifurcation $ }
		{
			x $ fixed point $.
			$ born of 2 fixed points $.
		}
	}
	
	
	\proposition{ $ Characterization of Pitchfork bifurcations $ }

		}

		{
			x $ Pitchfork $ \ifandonlyif.

			f(x) = x.

			f'(x) = 1.

			\partialderivative{ f }{ x^2 } = 0.

			\partialderivative{ f }{ \lambda } = 0
		}

		{

		}
	}
	
	
	\problem{ $ 9. Fixed points and 2-periodic points $ }

				f \convergesto 0.
			}


			\step{ $ Fixed points $}
			{
				f(x) = x \ifandonlyif x = \pm \sqrt{ r - 1 }.

				f'(\pm \sqrt{ r - 1 })  = \frac{ 2 - r }{ r }
			}

			\step{ $ n-periodic points $ }
			{
				f^n(x) = x 
			}
		}
