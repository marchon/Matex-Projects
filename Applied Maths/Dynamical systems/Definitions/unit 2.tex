\unit{ $ 2-D linear dynamical systems $ }
{
	
	\definition{ $ Linear system $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $
		}
		\then{ ( M, \N, f ) }{ $ linear $ }
		{
			\ex{ A \in \Mc_{n\times n}(\R) }
			{
				\definedFunction{ f }{ \R^n }{ \R^n }{ x }{ Ax }
			}
		}
	}


	\definition{ $ Multiplier $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $.
			x \in M
		}
		\name{ $multiplier of $ x }
		{
			Df(p)
		}
	}

}
