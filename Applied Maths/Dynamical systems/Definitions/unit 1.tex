\unit{ $ Unidimensional discrete dynamical systems $ }
{
	\introduction
	{ 
		$introduction\\$ 
	}

	\definition{ $ Dynamical system $ }
	{
		\letbe
		{
			M $ manifold $.
			T $ set $.
			\Phi : \family{ \Phi_t }{ t } \subset \functionSpace{ M }{ M }
		}
		\then{ ( M, T, \Phi ) }{ $ a dynamical system $ }
		{
			T \in \set{ \N, \Z, \R, \R^+ }
		}
		\denote
		{
			T \as $ time of $ ( M, T, \Phi )
		}
	}
	
	
	
	

}
