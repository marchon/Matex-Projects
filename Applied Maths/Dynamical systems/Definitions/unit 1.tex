\unit{ $ Discrete dynamical systems $ }
{	
	
	\definition{ $ Dynamical system $ }
	{
		\letbe
		{
			M $ manifold $.
			T $ monoid $.

			\function{ \phi }{ M \times T }{ M }
		}
		\then{ ( M, T, \phi ) }{ $ a dynamical system $ }
		{
			\all{ x \in X }
			{
				\phi( x, 0 ) = 0.

				\all{ t_1,t_2 \in T }
				{
					\phi( \phi( x, t_1 ), t_2 ) = \phi( x, t_1+t_2 )
				}
			}
		}

	}
	\definition{ $ Dimension $ }
	{
		\letbe
		{
			( M, T, \phi ) $ dynamical system $
		}
		\name{ $ dimension of $ ( M, T, \phi ) }
		{
			\dim(M)
		}
		\denote
		{
			dim(M) = n \as ( M, T, \phi ) $ n-D $
		}
	}


	\definition{ $ Discrete \& Continuous$ }
	{
		\letbe
		{
			( M, T, \phi ) $ dynamical system $
		}
		\then{ ( M, T, \phi ) }{ $ discrete $ }
		{
			T \countable
		}
		\then{ ( M, T, \phi ) }{ $ continuous $ }
		{
			T \subset \R.
			T $ open $
		}
	}


	\definition{ $ Defined by a function $ }
	{
		\letbe
		{
			( M, T, \phi ) $ dynamical system $.
			\function{ f }{ M }{ M }
		}
		\then{ ( M, T, \phi ) }{ $ a dynamical system defined by $ f $ $ }
		{
			T = \N.

			\definedFunction{ \phi }{ M \times \N }{ M }{ (x,n) }{ f^n(x) }
		}
		\denote
		{
			( M, T, \phi ) $ dynamical system defined by $ f \as ( M, \N, f ).
			f \in \Cc^n( M ) \as ( M, \N, f ) \hspace{5pt} \Cc^n 
		}
	}
	
	
	\definition{ $ Orbit $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $.
			x \in M
		}
		\name{ $ orbit of $ x }
		{
			\family*{ f^n(x) }{ n \in \N }
		}
		\denote
		{
			o(x)
		}
	}
	
	
	\definition{ $ Periodicity $ }
	{
		\letbe
		{
			( M, \N, f ) $ dynamical system $.
			x \in M.
			m \in \N
		}
		\then{ x }{ $ a m-periodic point $ }
		{
			f^m(x)=x
		}
		\denote
		{
			\set{ x \in M \with f(x) = x } \as \fixed(f)
		}
	}


	\definition{ $ Stability $ }
	{
		\letbe
		{
			\function{ f }{ \R^n }{ \R^n }.
			p \in \R^n $ m-periodic point $
		}
		\then{ p }{ $ stable $ }
		{
			\all{ \epsilon \in \R^+ }
			{
				\ex{ \delta \in \R^+ }
				{
					\all{ x \in B(p,\delta) }
					{
						\all{ n \in \N }
						{
							f^{nm}(x) \in B(p,\epsilon)
						}
					}
				}
			}
		}
		\then{ p }{ $ unstable $ }
		{
			p $ not stable $
		}
	}
	
	
	\definition{ $ Attractive \& Repulsive $ }
	{
		\letbe
		{
			\function{ f }{ \R^n }{ \R^n }.
			p \in \R^n $ m-periodic point $
		}
		\then{ p }{ $ attractive $ }
		{
			p $ stable $.
			\ex{ \epsilon \in \R^+ }
			{
				\all{ x \in B(p,\epsilon) }
				{
					f^{nm}(x) \convergesto p
				}
			}
		}
		\then{ p }{ $ repulsive $ }
		{
			p $ attractive by $ f^{-1}.
			\all{ \Uc \subset M }[ \Uc $ open $ \logicAnd x \in \Uc ]
			{
				\all{ x' \in \Uc }[ x' \neq x ]
				{
					\ex{ N \in \N }
					{
						\all{ n \in \N }[ n \geq N ]
						{
							f^{nm}(x') \nin \Uc
						}
					}
				}
			}
		}
	}
	
	
	
	\definition{ $ Fixed point character $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $
		}
		\name{ $ Fixed point character $ }
		{
			\definedFunction{ f }{ \fixed(f) }{ \set{ -1,0,1 } }{ x }{ \begin{cases} +1 & x $ repulsive $ \\ -1 & x $ attractive $ \\ 0 & x $ no attractive no repulsive $  \end{cases} }
		}
		\denote
		{
			f \as \chi_f
		}
	}
	
	
	\definition{ $ Attraction set $ }
	{
		\letbe
		{
			( M, \N, f ) $ dynamical system $.
			x \in M $ attractive m-periodic point $.
			o(x) $ orbit of $ x
		}
		\name{ $ attraction set of $ x }
		{
			\set{ y \in M \with \exists \; x' \in o(x) \as f^{nm}(y) \convergesto x' }
		}
		\denote
		{
			A( x )
		}
	}


	\definition{ $ Multiplier $ }
	{
		\letbe
		{
			( M, \N, f ) \hspace{5pt} \Cc^1 $ dynamical system $.
			x \in M
		}
		\name{ $ multiplier of $ x }
		{
			f'(x)
		}
		\denote
		{
			m(x).
			|m(x)|= 1 \as x $ neutral point $
		}
	}


	\definition{ $ Feeble point$ }
	{
		\letbe
		{
			( M, \N, f ) \hspace{5pt} \Cc^3 $ dynamical system $.
			x \in M
		}
		\then{ x }{ $ feeble point $ }
		{
			x $ neutral point $.
			f''(x) = 0
		}
	}
	

	\definition{ $ Sarkovskii's order $ }
	{
		\name{ $ Sarkovskii's order	$ }
		{
			a = 2^na', b = 2^mb'.
			a <_s b \ifandonlyif 
			\begin{cases} 
			m < n & a' = b' = 1 \\ 
			* & a' = 1, b' \neq 1 \\ 
			a' < b' & a'=b'\neq 1 \\ 
			n < m & 1 \neq a' \neq b'
			\end{cases}
		}
	}
	
	
	\definition{ $ Chaos $ }
	{
		\letbe
		{
			( \R, \N, f ) $ dynamical system $
		}
		\then{ ( \R, \N, f ) }{ $ chaotic $ }
		{
			\fixed(f) $ dense in $ \R.
			\ex{ x \in \R }
			{
				o(x) $ dense in $ \R
			}.

			\all{ x \in \R }
			{
				\ex{ \epsilon \in \R^+ }
				{
					\all{ \delta \in \R^+ }
					{
						\ex{ \tilde{x} \in B(x,\delta) }
						{
							\limit{ o(\tilde{x}) }{ n } \nin B(x,\epsilon)
						}
					}
				}
			}
		}
	}	
	
	
	\definition{ $ Topological equivalence $ }
	{
		\letbe
		{
			( M, \N, f_1 ), ( M, \N, f_2' ) $ dynamical systems $
		}
		\then{ ( M, \N, f_1 ) }{ $ topologically equivalent to $ ( M, \N, f_2 ) $ $ }
		{
			\fixed(f) = \fixed(f').

			\all{ x \in \fixed(f) }
			{
				\chi_{f_1}(x) = \chi_{f_2}(x)
			}
		}
		\denote
		{
			( M, \N, f ) \sim ( M, \N, f' )
		}
	}
	
	
	\definition{ $ Bifurcation $ }
	{
		\letbe
		{
			\Lambda \subset M.
			( M, \N, f_\lambda )_{\lambda \in \Lambda} $ dynamical systems parametrized by $ \Lambda.
			\lambda_0 \in \Lambda
		}
		\then{ \lambda_0 }{ $ a bifurcation value $ }
		{
			\all{ \epsilon \in \R^+ }
			{
				\ex{ \lambda_1,\lambda_2 \in B(\lambda_0,\epsilon) }
				{
					( M, \N, f_{\lambda_1} ) \not\sim ( M, \N, f_{\lambda_2} )
				}
			}
		}
	}
	
	
	\definition{ $ Saddle-node bifurcation $ }
	{
		\letbe
		{
			\Lambda \subset M.
			( M, \N, f_\lambda )_{\lambda \in \Lambda} $ dynamical systems parametrized by $ \Lambda.
			\lambda_0 \in \Lambda.
			x_0 \in M
		}
		\then{ \lambda_0 }{ $ a saddle-node bifurcation value at $ x_0 $ $ }
		{
			x_0 \in \fixed( f_{\lambda_0} ).
			x_0 $ neutral point of $ f_{\lambda_0}.
			\partialderivative{ f_\lambda }{ \lambda }(x_0) \neq 0.
			\partialderivative{ f_\lambda }{ xx }(x_0) \neq 0
		}
	}
	
	
	\definition{ $ Pitchfork bifurcation $ }
	{
		\letbe
		{
			\Lambda \subset M.
			( M, \N, f_\lambda )_{\lambda \in \Lambda} $ dynamical systems parametrized by $ \Lambda.
			\lambda_0 \in \Lambda.
			x_0 \in M
		}
		\then{ \lambda_0 }{ $ Pitchfork bifurcation value at $ x_0 $ $ }
		{
			x_0 \in \fixed( f_{\lambda_0} ).
			x_0 $ neutral point of $ f_{\lambda_0}.
			\partialderivative{ f_\lambda }{ \lambda }(x_0,\lambda_0) = 0.
			\partialderivative{ f_\lambda }{ x^2 }(x_0,\lambda_0) = 0.
			\partialderivative{ f_\lambda }{ \lambda x }(x_0,\lambda_0) \neq 0.
			\partialderivative{ f_\lambda }{ x^3 }(x_0,\lambda_0) \neq 0
		}
	}
	
	
	\definition{ $ Period doubling bifurcation $ }
	{
		\letbe
		{
			\Lambda \subset M.
			( M, \N, f_\lambda )_{\lambda \in \Lambda} $ dynamical systems parametrized by $ \Lambda.
			\lambda_0 \in \Lambda.
			x_0 \in M
		}
		\then{ \lambda_0 }{ $ Period doubling bifurcation value at $ x_0 $ $ }
		{
			\lambda_0 $ Pitchfork bifurcation value at $ x_0 $ of $ f^2
		}
	}	

}
