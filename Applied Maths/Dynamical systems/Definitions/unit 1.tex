\unit{ $ One-dimensional discrete dynamical systems $ }
{	
	
	
	\definition{ $ Dynamical system $ }
	{
		\letbe
		{
			M $ manifold $.
			T $ monoid $.

			\function{ \phi }{ M \times T }{ M }
		}
		\then{ ( M, T, \phi ) }{ $ a dynamical system $ }
		{
			\all{ x \in X }
			{
				\phi( x, 0 ) = 0.

				\all{ t_1,t_2 \in T }
				{
					\phi( \phi( x, t_1 ), t_2 ) = \phi( x, t_1+t_2 )
				}
			}
		}

	}
	\definition{ $ Dimension $ }
	{
		\letbe
		{
			( M, T, \phi ) $ dynamical system $
		}
		\name{ $ dimension of $ ( M, T, \phi ) }
		{
			\dim(M)
		}
		\denote
		{
			dim(M) = n \as ( M, T, \phi ) $ n-D dynamical system $
		}
	}


	\definition{ $ Discrete \& Continuous$ }
	{
		\letbe
		{
			( M, T, \phi ) $ dynamical system $
		}
		\then{ ( M, T, \phi ) }{ $ discrete $ }
		{
			T \countable
		}
		\then{ ( M, T, \phi ) }{ $ continuous $ }
		{
			T \subset \R.
			T $ open $
		}
	}


	\definition{ $ Defined by a function $ }
	{
		\letbe
		{
			( M, T, \phi ) $ dynamical system $.
			\function{ f }{ M }{ M }
		}
		\then{ ( M, T, \phi ) }{ $ a dynamical system defined by $ f $ $ }
		{
			T = \N.

			\definedFunction{ \phi }{ M \times \N }{ M }{ (x,n) }{ f^n(x) }
		}
		\denote
		{
			( M, T, \phi ) $ dynamical system defined by $ f \as ( M, \N, f ).
			f \in \Cc^n( M ) \as ( M, \N, f ) \hspace{5pt} \Cc^n $ dynamical system $
		}
	}
	
	
	\definition{ $ Orbit $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $.
			x \in M
		}
		\name{ $ orbit of $ x }
		{
			\family*{ f^n(x) }{ n \in \N }
		}
		\denote
		{
			o(x)
		}
	}
	
	
	\definition{ $ n-periodic point $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $.
			x \in M.
			n \in \N
		}
		\then{ x }{ $ a n-periodic point $ }
		{
			f^n(x) = x.
			\all{ n' \in \N }[ n' < n ]
			{
				f^{n'}(x) \neq x
			}
		}
		\denote
		{
			\set{ x \in M \with f(x) = x } \as Fix(f)
		}
	}
	

	\definition{ $ Attractive \& Repulsive $ }
	{
		\letbe
		{
			( M, \N, f ) $ metrical dynamical system $.
			x \in M $ m-periodic point $
		}
		\then{ x }{ $ attractive $ }
		{
			\ex{ \Uc \in M }
			{
				\Uc $ open $.

				\all{ x' \in \Uc }
				{
					\ex{ N \in \N }
					{
						\all{ n \in \N }[ n \geq N ]
						{
							f^{nm}(x') \in \Uc
						}
					}
				}
			}
		}
		\then{ x }{ $ repulsive $ }
		{
			\all{ \Uc \subset M }[ \Uc $ open $ \logicAnd x \in \Uc ]
			{
				\all{ x' \in \Uc }[ x' \neq x ]
				{
					\ex{ N \in \N }
					{
						\all{ n \in \N }[ n \geq N ]
						{
							f^{nm}(x') \nin \Uc
						}
					}
				}
			}
		}
	}
	
	
	\definition{ $ Fixed point character $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $
		}
		\name{ $ Fixed point character $ }
		{
			\definedFunction{ f }{ Fix(f) }{ \set{ +,- } }{ x }{ \begin{cases} + & x $ repulsive $ \\ - & x $ attractive $ \end{cases} }
		}
		\denote
		{
			f \as \chi
		}
	}
	
	
	\definition{ $ Attraction set $ }
	{
		\letbe
		{
			( M, \N, f ) $ dynamical system $.
			x \in M $ attractive m-periodic point $.
			o(x) $ orbit of $ x
		}
		\name{ $ attraction set of $ x }
		{
			\set{ y \in M \with \exists \; x' \in o(x) \as f^{nm}(y) \convergesto x' }
		}
		\denote
		{
			A( x )
		}
	}
	
	
	\definition{ $ Neutral point $ }
	{
		\letbe
		{
			( M, \N, f ) $ differentiable dynamical system $.
			x \in M
		}
		\then{ x }{ $ a neutral point $ }
		{
			f'(x) \in \set{ -1,1 }
		}
	}


	\definition{ $ Feeble point $ }
	{
		\letbe
		{
			( M, \N, f ) \hspace{5pt} \Cc^3 $ dynamical system $.
			x \in M $ neutral point $
		}
		\then{ x }{ $ feeble point $ }
		{
			f''(x) = 0
		}
	}
	

	\definition{ $ Saddle point $ }
	{
		\letbe
		{
			\Uc \subset \R^n.
			f \in \Cc^1(\Uc).
			x \in \Uc
		}
		\then{ x }{ $ a saddle point $ }
		{
			f'(x) = 0
		}
	}

	
	\definition{ $ Homeomorphism $ }
	{
		\letbe
		{
			( X_1, \tau_1 ), ( X_2, \tau_2 ) $ topological spaces $.
			\function{ f }{ X_1 }{ X_2 }
		}
		\then{ f }{ $ a homeomorphism $ }
		{
			f $ biyective $.
			f \in \Cc( X_1 ).
			f^{-1} \in \Cc( X_2 )			
		}
		\denote
		{
			\set{ \function{ f }{ X_1 }{ X_2 } \with f $ homeomorphism $ } \as Homeo( X_1 )
		}
	}
	
	
	
	
	\definition{ $ Multiplier $ }
	{
		\letbe
		{
			( M, \N, f ) \Cc^1 $ dynamical system $.
			x \in M
		}
		\name{ $ multiplier of $ x }
		{
			f'(x)
		}
	}
	
	
	\definition{ $ Logistic $ }
	{
		\letbe
		{
			\function{ f }{ \R }{ \R }.
			( M, T, \phi ) $ dynamical system defined by $ f
		}
		\then{ ( M, T, \phi ) }{ $ a logistic dynamical system $ }
		{
			\ex{ a \in \R }
			{
				\definedFunction{ f }{ \R }{ \R }{ x }{ ax(1-x) }
			}
		}
	}
	
	
	\definition{ $ Chaos $ }
	{
		\letbe
		{
			\function{ f }{ \R }{ \R }.
			( M, T, \phi ) $ dynamical system defined by $ f
		}
		\then{ ( M, T, \phi ) }{ $ chaotic $ }
		{
			Fix(f) $ dense in $ \R.
			\ex{ x \in \R }
			{
				o(x) $ dense in $ \R
			}.

			f $ sensibility of $ x_0
		}
	}


	\definition{ $ Sarkovskii's order $ }
	{
		\name{ $ Sarkovskii's order	$ }
		{
			a = 2^na', b = 2^mb'.
			a < b \ifandonlyif \begin{cases} m < n & a' = b' = 1 \\ * & a' = 1, b' \neq 1 \\ a' < b' & a'=b'\neq 1 \\ n < m & 1 \neq a' \neq b'\end{cases}
		}
		\denote
		{
			a < b \as a \prec b
		}
	}
	
	
	
	
	
	\definition{ $ Topologically equivalent $ }
	{
		\letbe
		{
			( M, \N, f ) $ functional dynamical system $.
			( M, \N, f' ) $ functional dynamical system $
		}
		\then{ ( M, \N, f ) }{ $ topologically equivalent to $ ( M, \N, f' ) }
		{
			Fix(f) = Fix(f').

			\all{ x \in Fix(f) }
			{
				character_f(x) = character_{f'}(x)
			}
		}
		\denote
		{
			( M, \N, f ) ~ ( M, \N, f' )
		}
	}
	
	
	\definition{ $ Bifurcation point $ }
	{
		\letbe
		{
			( M, \N, f_\lambda )_{\lambda \in \Lambda} $ functional dynamical system family $.
			\lambda' \in \Lambda
		}
		\then{ \lambda' }{ $ a bifurcation value $ }
		{
			\all{ \epsilon \in \R^+ }
			{
				\ex{ \lambda'' \in (\lambda' - \epsilon, \lambda' + \epsilon ) }
				{
					( M, \N, f_{\lambda''} ) $ not topologically equivalent to $ ( M, \N, f_{\lambda'} )
				}
			}
		}
	}

	{
		\letbe
